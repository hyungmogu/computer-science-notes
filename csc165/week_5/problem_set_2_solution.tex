\section*{Question 1}
\begin{enumerate}[a.]
    \item

    \textbf{Fact 1:}

    \bigskip

    Let $a,b \in \mathbb{N}$. Assume $2 \nmid a$ and $2 \nmid b$.

    \bigskip

    Then, $\exists k,l \in \mathbb{Z}$.

    \begin{align}
        a &= 2k - 1\\
        b &= 2l - 1
    \end{align}

    \bigskip

    Then,

    \begin{align}
        a - b &= (2k-1)(2l-1)\\
        &= 2(k-1)-2\\
        &= 2(k-l-1)
    \end{align}

    \bigskip

    Then, it follows from the definition of divisibility that $2 \mid a-b$.
\end{enumerate}
