\begin{enumerate}[a.]
    \item

    $P(n): \forall n \in \mathbb{N},\: n \leq 2^{n}$.

    $\forall k \in \mathbb{N}, P(0) \land P(k) \Rightarrow P(k+1)$

    \bigskip

    Or, with $P$ fully expanded,

    $\forall k \in \mathbb{N},\:0 \leq 2^0 \land k \leq 2^k \Rightarrow k+1 \leq
    2^{k+1}$
\end{enumerate}
