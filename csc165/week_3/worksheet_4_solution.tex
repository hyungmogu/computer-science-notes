\section*{Question 1}
\begin{enumerate}[a.]
    \item $\exists n \in \mathbb{N}, (n > 3) \land (n^2-1.5n \geq 5)$
    \item The variable is existentially quantified
    \item Concrete natural number
    \item

    Let n = 5.

    Then,

    \begin{equation}
        (5)^2 - 1.5(5)
    \end{equation}

    Then,

    \begin{equation}
        25 - 7.5
    \end{equation}

    Then,

    \begin{equation}
        17.5
    \end{equation}

    which is greater than 5, so the statement is True


\end{enumerate}
