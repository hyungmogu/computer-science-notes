\section*{Question 1}

\begin{enumerate}[a)]
    \item
        A = \{\,2, 5\,\}

        $A^c$ = \{\,1, 3, 4, 6\,\}

    \bigskip

    \item
        $A^c$ = $U \setminus A$

    \bigskip

    \item
        $A^c \cap B^c$ = \{\,$x \mid x \in U$, $x \leq 0$ and $x \geq 4$\,\}

        $A^c \cap B^c$ = \{\,$x \mid x \in U$, $x < 1$ and $x > 2$\,\}

        $(A \cap B)^c$ = \{\,$x \mid x \in U$, $x < 1$ and $x > 2$\,\}

        $(A \cup B)^c$ = \{\,$x \mid x \in U$, $x \leq 0$ and $x \geq 4$\,\}
\end{enumerate}

\section*{Question 4}

\begin{enumerate}[a)]
    \item
        \begin{tabular}{ c | c | c }
            $x$ & $\lfloor x \rfloor$ & $\lceil x \rceil$ \\
            \hline
            $\frac{25}{4}$ & 6 & 7 \\
            \hline
            0.999 & 0 & 1 \\
            \hline
            -2.01 & -3 & -2 \\
            \hline
        \end{tabular}


    \bigskip

    \item
        Domain: $\mathbb{R}$

        Codomain: $\mathbb{Z}$

    \bigskip

    \item

        $1 =  \lfloor 0.75 + 0.25 \rfloor$
