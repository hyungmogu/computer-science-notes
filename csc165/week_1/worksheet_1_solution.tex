    \item
        A = \{\,2, 5\,\}

        $A^c$ = \{\,1, 3, 4, 6\,\}

    \bigskip

    \item
        $A^c$ = $U \setminus A$

    \bigskip

    \item
        $A^c \cap B^c$ = \{\,$x \mid x \in U$, $x \leq 0$ and $x \geq 4$\,\}

        $A^c \cap B^c$ = \{\,$x \mid x \in U$, $x < 1$ and $x > 2$\,\}

        $(A \cap B)^c$ = \{\,$x \mid x \in U$, $x < 1$ and $x > 2$\,\}

        $(A \cup B)^c$ = \{\,$x \mid x \in U$, $x \leq 0$ and $x \geq 4$\,\}
\end{enumerate}

            \hline

            \hline

            \hline
