    {
        struct node *cur = list, *prev = NULL;

        if (list == NULL) {
            list = new_node;
            return list;
        }

        while (cur != NULL && cur->value <= new_node ->value) {
            prev = cur;
            cur = cur->next;
        }

        prev->next = new_node;
        new_node->next = cur;
        return list;
