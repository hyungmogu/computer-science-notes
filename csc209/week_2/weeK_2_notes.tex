    \end{lstlisting}

    \item \textit{read}
    \begin{itemize}
    \item store values in variable interactively``

    \begin{lstlisting}[language=bash]
    >>> read x y
    >>> hello world 2
    >>> echo $x
    hello
    >>> echo $y
    world 2
    \end{lstlisting}

    \end{itemize}
    \item \textit{``}
    \begin{itemize}
    \item store commands in variable

    \begin{lstlisting}[language=bash]
    >>> i = `expr 4 + 1`
    >>> echo $i
    5
    \end{lstlisting}
    \end{itemize}
\end{itemize}
